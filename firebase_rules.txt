# Regras do Firebase Realtime Database
# Acesse o console do Firebase (https://console.firebase.google.com/)
# Navegue até o Realtime Database > Regras
# Cole o código abaixo e clique em "Publicar"

{
  "rules": {
    ".read": true,  // Permite leitura sem autenticação
    "scores": {
      ".read": true,  // Permite leitura de pontuações sem autenticação
      ".write": "auth != null",  // Permite escrita apenas para usuários autenticados
      "$score": {
        ".validate": "newData.hasChildren(['user_id', 'name', 'score', 'timestamp'])"
      }
    },
    "users": {
      "$user_id": {
        ".read": "auth != null && auth.uid == $user_id",
        ".write": "auth != null && auth.uid == $user_id"
      }
    }
  }
}

# Regras alternativas (mais permissivas - use apenas para teste)
{
  "rules": {
    ".read": true,
    ".write": true
  }
}
